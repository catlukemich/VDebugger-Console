`
from vdebugger import *


def test_console():
    vd("a", "B")
    vd("a", "B")
    vd("a", "B")
    vd("b", "EEE")
    vd("somevar", 3141)
    vd("foo", 123)
    vd("hoo", {"nope": "yes"})
    vd("blu", {"nope": "nonono"})
    vd("blu", {"nope": "nonono"})
`
